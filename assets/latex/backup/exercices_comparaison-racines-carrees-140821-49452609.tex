\meta{titre}{Comparaison de racines carrées}
\meta{type}{exercices}
%
Compléter à l'aide du signe $ > $ ou $ < $, en indiquant la propriété utilisée.
\begin{enumerate}
     \item
     $\sqrt{2}$ ... $\sqrt{3}$
     \item
     $\sqrt{5}$ ... $2$
     \item
     $\sqrt{3}$ ... $-2$
     \item
     $-\sqrt{7}$ ... $-3$
     \item
     $\frac{\sqrt{5}}{2}$ ... $1$
\end{enumerate}
