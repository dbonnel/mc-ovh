%##
\begin{enumerate}
     \item
     Montrer que la droite $\left(MN\right)$ est parallèle à la droite $\left(BD\right)$
     \item
     Construire (en justifiant la construction) l'intersection du plan $\left(AMN\right)$ avec le plan $\left(ABCD\right)$
     \item
     Déterminer et construire l'intersection du plan $\left(AMN\right)$ avec la droite $\left(DC\right)$
     \item
     Construire (en justifiant la construction) la trace du plan $\left(AMN\right)$ sur la face $CDHG$
     \item
     Tracer la section du cube $ABCDEFGH$ par le plan $\left(AMN\right)$
\end{enumerate}
