\begin{center}

\begin{enumerate}
     \item
     $x \mapsto  \frac{1}{2}x-1$
     \item
     $x \mapsto  -x+3$
     \item
     $x \mapsto  2$
     \item
     $x \mapsto  2x+1$
     \item
     $x \mapsto  -2x+1$
\end{enumerate}
