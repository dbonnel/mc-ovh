%
Donner le tableau de variations de la fonction $f$ définie sur $\left[-1 ; 3\right]$ dont la courbe est représentée ci-dessous.

\\
\begin{center}
     \begin{extern}%style="width:25rem" alt="représentation graphique de la fonction"
           \resizebox{8cm}{!}{%
               \def\xmin{-1.5}
               \def\xmax{3.5}
               \def\ymin{-1.9}
               \def\ymax{3.5}
               \def\xunit{1}  % unités en cm 
               \def\yunit{1}
               \psset{xunit=\xunit,yunit=\yunit,algebraic=true}
               \fontsize{12pt}{12pt}\selectfont
               \begin{pspicture*}[linewidth=1pt](\xmin,\ymin)(\xmax,\ymax) 
                    \psaxes[linewidth=0.5pt]{->}(0,0)(\xmin,\ymin)(\xmax,\ymax) 
                    \rput[tr](-0.2,-0.3){$O$} 
                    \rput[tl](-1.5,1.5){$\blue\mathscr{C}_f$} 
                    \psplot[plotpoints=1000,linewidth=0.8pt,linecolor=blue]{-1}{3}{  3*x*x - 1*x*x*x - 1} 
               \end{pspicture*}
          }
     \end{extern} 
\end{center}
\\
\textit{(On considèrera que les coordonnées des extremums sont entières.)}
\begin{corrige}
\begin{center}
\begin{extern}%width="450" alt="tableau de variations de la fonction"
                    \begin{tikzpicture}[scale=0.875]
                         % Styles
                         \tikzstyle{cadre}=[thin]
                         \tikzstyle{fleche}=[->,>=latex,thin]
                         \tikzstyle{nondefini}=[lightgray]
                         % Dimensions Modifiables
                         \def\Lrg{1.5}
                         \def\HtX{1}
                         \def\HtY{0.5} 
                         % Dimensions Calculées
                         \def\lignex{-0.5*\HtX}
                   
                         \def\separateur{-0.5*\Lrg}
                         % Largeur du tableau
                         \def\gauche{-1.5*\Lrg} 
                         \def\droite{6.5*\Lrg}
                         % Hauteur du tableau
                         \def\haut{0.5*\HtX}
                         \def\bas{-2*\HtX-\HtY} 
                         % Pointillés
                         \draw[dotted,black] (0*\Lrg,\lignex) -- (0*\Lrg,\bas); 
                         \draw[dotted,black] (2*\Lrg,\lignex) -- (2*\Lrg,\bas);
                         \draw[dotted,black] (4*\Lrg,\lignex) -- (4*\Lrg,\bas);
                         \draw[dotted,black] (6*\Lrg,\lignex) -- (6*\Lrg,\bas);
                         % Ligne de l'abscisse~: x
                         \node at (-1*\Lrg,0) {$x$};
                         \node at (0*\Lrg,0) {$-1$};
                         \node at (2*\Lrg,0) {$0$};
                         \node at (4*\Lrg,0) {$2$}; 
                         \node at (6*\Lrg,0) {$3$};
                         % Ligne de la fonction~: f(x)
                         \node  at (-1*\Lrg,{-1*\HtX+(-1)*\HtY}) {$f(x)$};
                         \node (f1) at (0*\Lrg,{-1*\HtX+(0)*\HtY}) {$3$};
                         \node (f2) at (2*\Lrg,{-1*\HtX+(-2)*\HtY}) {$-1$};
                         \node (f3) at (4*\Lrg,{-1*\HtX+(0)*\HtY}) {$3$};
                         \node (f4) at (6*\Lrg,{-1*\HtX+(-2)*\HtY}) {$-1$};
                         % Flèches
                         \draw[fleche] (f1) -- (f2);
                         \draw[fleche] (f2) -- (f3);
                         \draw[fleche] (f3) -- (f4);
                         % Encadrement
                         \draw[cadre] (\separateur,\haut) -- (\separateur,\bas);
                         \draw[cadre] (\gauche,\haut) rectangle  (\droite,\bas);
                         \draw[cadre] (\gauche,\lignex) -- (\droite,\lignex);
                         \draw[cadre] (\gauche,\lignex) -- (\droite,\lignex);
                    \end{tikzpicture}
                              \end{extern}
          \end{center}

\end{corrige}
