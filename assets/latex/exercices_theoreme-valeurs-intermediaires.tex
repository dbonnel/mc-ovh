\begin{enumerate}
     \item
     Montrer que l'équation :

\begin{center}
     $\frac{2x+3}{x+1}=x^2$
\end{center}

     admet une unique solution sur l'intervalle $[1;2]$
     \item
     À l'aide d'une calculatrice donner une valeur approchée de cette solution à $10^{-3}$ près
\end{enumerate}
